============
 prometheus
============

cd /opt
wget https://github.com/prometheus/prometheus/releases/download/v2.1.0/prometheus-2.1.0.linux-amd64.tar.gz
tar xzvf prometheus-2.1.0.linux-amd64.tar.gz
cd prometheus-2.1.0.linux-amd64
nano prometheus.yml
  ---

  global:
    scrape_interval:     10s
    evaluation_interval: 10s
    external_labels:
        monitor: 'blak'

  scrape_configs:
    - job_name: 'prometheus'
      static_configs:
        - targets: ['blak0:9090']
    - job_name: 'node'
      static_configs:
        - targets:
          - prmths0:9100
          - vis0:9100
          - nginx0:9100

nano /lib/systemd/system/prometheus.service
  [Unit]
  Description=prometheus
  After=network.target

  [Service]
  User=root
  ExecStart=/opt/prometheus-2.1.0.linux-amd64/prometheus --storage.tsdb.retention=365d
  WorkingDirectory=/opt/prometheus-2.1.0.linux-amd64
  Restart=on-failure

  [Install]
  WantedBy=multi-user.target

systemctl enable prometheus
systemctl daemon-reload
service prometheus start

rm prometheus-2.1.0.linux-amd64.tar.gz
